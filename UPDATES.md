-- =========================================================
-- TABLA: persons
-- =========================================================
CREATE TABLE persons (
    id                 TEXT PRIMARY KEY,
    nombre             TEXT NOT NULL,
    telefono           TEXT,
    correo_electronico  TEXT UNIQUE,
    direccion          TEXT,
    referencias        TEXT
);

-- =========================================================
-- TABLA: shipments
-- =========================================================
CREATE TABLE shipments (
    id_envio            BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    sender_id           TEXT NOT NULL,
    recipient_id        TEXT NOT NULL,
    tipo_servicio       TEXT NOT NULL,
    nivel_prioridad     INTEGER NOT NULL DEFAULT 3,
    codigo_rastreo      TEXT NOT NULL UNIQUE,
    estado_actual       TEXT NOT NULL,
    fecha_envio         DATE NOT NULL,
    fecha_estimada      DATE,
    fecha_actualizacion DATE,
    costo_total         NUMERIC(12,2) NOT NULL DEFAULT 0,
    instrucciones_envio TEXT,
    CONSTRAINT fk_shipments_sender
        FOREIGN KEY (sender_id) REFERENCES persons(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,
    CONSTRAINT fk_shipments_recipient
        FOREIGN KEY (recipient_id) REFERENCES persons(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,
    CONSTRAINT chk_nivel_prioridad
        CHECK (nivel_prioridad BETWEEN 1 AND 5),
    CONSTRAINT chk_costo_total
        CHECK (costo_total >= 0)
);

-- =========================================================
-- TABLA: packages
-- =========================================================
CREATE TABLE packages (
    id_paquete   BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    shipment_id  BIGINT NOT NULL,
    peso         NUMERIC(10,2) NOT NULL,
    largo        NUMERIC(10,2),
    ancho        NUMERIC(10,2),
    alto         NUMERIC(10,2),
    descripcion  TEXT,
    CONSTRAINT fk_packages_shipment
        FOREIGN KEY (shipment_id) REFERENCES shipments(id_envio)
        ON UPDATE CASCADE
        ON DELETE CASCADE,
    CONSTRAINT chk_peso
        CHECK (peso > 0)
);

-- =========================================================
-- TABLA: logistic_events
-- =========================================================
CREATE TABLE logistic_events (
    id_evento        BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    shipment_id      BIGINT NOT NULL,
    tipo_evento      TEXT NOT NULL,
    estado_resultante TEXT NOT NULL,
    ubicacion        TEXT,
    observacion      TEXT,
    fecha_evento     TIMESTAMP NOT NULL,
    CONSTRAINT fk_logistic_events_shipment
        FOREIGN KEY (shipment_id) REFERENCES shipments(id_envio)
        ON UPDATE CASCADE
        ON DELETE CASCADE
);
